let arquivoRoupaLocal = null; let linkFinalGlobal = ''; function atualizarFotoRoupa(input) { if (input.files && input.files[0]) { arquivoRoupaLocal = input.files[0]; document.getElementById('fotoRoupaUrl').value = ''; const status = document.getElementById('statusRoupa'); status.innerText = '✅ Foto selecionada: ' + input.files[0].name; status.style.display = 'block'; } } function limparArquivoLocal() { arquivoRoupaLocal = null; document.getElementById('statusRoupa').style.display = 'none'; } function comprimirImagem(file, maxWidth = 1024, quality = 0.8) { return new Promise((resolve, reject) => { const reader = new FileReader(); let height = img.height; if (width > maxWidth) { height = Math.round((height * maxWidth) / width); width = maxWidth; } canvas.width = width; canvas.height = height; const ctx = canvas.getContext('2d'); ctx.drawImage(img, 0, 0, width, height); resolve(canvas.toDataURL('image/jpeg', quality)); }; }; }); } async function gerarLinkOferta() { const btn = document.getElementById('btnGerar'); const loja = document.getElementById('nomeLoja').value.trim(); const validadeMinutos = parseInt(document.getElementById('validadeSelect').value); let urlRoupaFinal = ''; btn.disabled = true; btn.innerText = '⏳ Enviando foto...'; try { if (urlInput !== '') { urlRoupaFinal = urlInput; } else if (arquivoRoupaLocal) { const base64Comprimido = await comprimirImagem(arquivoRoupaLocal); const res = await fetch('/api/upload', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ image: base64Comprimido }) }); const data = await res.json(); if (data.sucesso && data.url) { urlRoupaFinal = data.url; } else { alert('Erro no envio da foto: ' + (data.erro || 'Falha no servidor.')); btn.disabled = false; btn.innerText = '⚡ Gerar Link ProvouComprou'; return; } } else { alert('Por favor, informe a URL da foto da roupa ou tire uma foto!'); btn.disabled = false; btn.innerText = '⚡ Gerar Link ProvouComprou'; return; } const expiraEm = Date.now() + (validadeMinutos * 60 * 1000); // Usa a raiz do domínio (sem /index.html) e variáveis de 1 letra: // l = loja, r = roupa, c = categoria, k = checkout, e = expira const baseUrl = window.location.origin; const linkFinal = `${baseUrl}/?l=${encodeURIComponent(loja)}&r=${encodeURIComponent(urlRoupaFinal)}&c=${categoria}&k=${encodeURIComponent(linkCupom)}&e=${expiraEm}`;
